        ascii = ascii-32
        camel_str = camel_str + chr(ascii)
    elif ascii >= 65 and ascii<=90:
        ascii = ascii + 32

vowel_sum = 0
vowel = {"a":1,"e":2,"i":3,"o":4,"u":5}
for x in string:
    for k,v in vowel.items():
        if x == k:
            vowel_sum = vowel_sum + v
print(vowel_sum)
